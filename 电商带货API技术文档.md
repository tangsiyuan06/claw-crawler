#### 关键指标监控阈值

| 指标 | 预警值 | 行动阈值 | 应对措施 |
|------|--------|----------|----------|
| 在线人数 | <100 | <50 | 加大互动/发福利 |
| 转化率 | <1% | <0.5% | 调整话术/优化商品 |
| 库存预警 | <20% | <10% | 紧急补货/下架 |
| 客诉率 | >2% | >5% | 暂停销售/客服介入 |
| 网络延迟 | >500ms | >1000ms | 切换网络/降低码率 |

#### 订单状态流转

| 状态 | 说明 | 触发条件 | 下一步操作 |
|------|------|----------|------------|
| pending | 待付款 | 用户下单 | 等待付款/超时取消 |
| paid | 已付款 | 支付成功 | 订单审核 |
| processing | 处理中 | 审核通过 | 仓库发货 |
| shipped | 已发货 | 物流揽收 | 物流跟踪 |
| delivered | 已送达 | 用户签收 | 确认收货 |
| completed | 已完成 | 确认收货 | 订单归档 |
| refunded | 已退款 | 退款完成 | 订单关闭 |
| cancelled | 已取消 | 取消订单 | 订单关闭 |

  }

// 指数退避重试
async function requestWithRetry(apiCall, maxRetries = 3) {
  let delay = 1000;
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await apiCall();
    } catch (error) {
      if (error.status === 429 && i < maxRetries - 1) {
        await sleep(delay);
        delay *= 2; // 指数退避
      } else {
        throw error;
      }
    }
  }
}

// 错误处理中间件
function errorHandler(error) {
  const errorCode = error.code;
  const message = errorCodes[errorCode] || '未知错误';
  
  // 记录日志
  logError({
    timestamp: new Date(),
    errorCode,
    message,
    stack: error.stack
  });
  
  // 告警通知（严重错误）
  if ([500, 502, 503].includes(errorCode)) {
    sendAlert({
      level: 'critical',
      message: `系统错误：${message}`,
      errorCode
    });
  }
  
  return {
    success: false,
    code: errorCode,
    message
  };
}

```javascript
// Agent 库存同步服务
class InventorySyncAgent {
  async syncInventory() {
    const products = await this.getActiveProducts();
    
    for (const product of products) {
      // 获取供应商库存
      const supplierStock = await api.get('/api/supplier/stock/query', {
        supplierId: product.supplierId,
        offerId: product.supplierOfferId
      });
      
      // 计算可销售库存
      const availableStock = this.calculateAvailableStock(
        supplierStock.quantity,
        product.reservedStock
      );
      
      // 更新销售平台库存
      await api.post('/api/product/stock/update', {
        productId: product.id,
        stock: availableStock
      });
      
      // 检查预警
      await this.checkStockAlert(product, availableStock);
    }
  }
}
```

```javascript
async function autoRefundApproval(refundRequest) {
  // 检查是否符合自动通过条件
  const autoApproveConditions = [
    refundRequest.amount <= 100,  // 金额≤100 元
    refundRequest.reason === 'seven_day_no_reason',  // 七天无理由
    refundRequest.orderValue <= 500,  // 订单金额≤500 元
    refundRequest.userCreditScore >= 700,  // 用户信用分≥700
    refundRequest.returnCount <= 3  // 历史退货次数≤3
  ];
  
  const shouldAutoApprove = autoApproveConditions.every(c => c);
  
  if (shouldAutoApprove) {
    await api.post('/api/afterSale/refund/approve', {
      refundId: refundRequest.id,
      autoApproved: true,
      reason: '符合自动通过条件'
    });
    return { approved: true, auto: true };
  }
  
  // 转人工审核
  return { approved: false, auto: false, reason: '转人工审核' };
}
```

```javascript
class OrderFulfillmentAgent {
  constructor() {
    this.capabilities = [
      'orderProcessing',
      'supplierOrdering',
      'logisticsTracking',
      'exceptionHandling'
    ];
  }
  
  async processOrder(order) {
    try {
      // 1. 订单验证
      await this.validateOrder(order);
      
      // 2. 供应商下单
      const supplierOrder = await this.placeSupplierOrder(order);
      
      // 3. 物流跟踪
      await this.trackLogistics(supplierOrder);
      
      // 4. 签收确认
      await this.confirmDelivery(supplierOrder);
      
      return { success: true, orderId: order.id };
    } catch (error) {
      await this.handleException(order, error);
      return { success: false, error: error.message };
    }
  }
}
```
